
\section{Conjunts i relacions}
\iniciTest{t1}{d,b,a,c,b,d,d,b,a,c}

\begin{enumerate}
\pregunta Donat el conjunt $A=\left\{ -1,0,1\right\} $, quin dels següents
conjunts coincideix amb $A$?

\begin{enumerate}
\opcio{a}{$\left\{ x\in\mathbb{N}:x^{3}-x=0\right\} $
}
\opcio{b}{$\left\{ x\in\mathbb{Q}:x^{2}\leq1\right\} $
}
\opcio{c}{$\left\{ x\in\mathbb{R}:x^{2}-1=0\right\} $
}
\opcio{d}{$\left\{ x\in \mathbb{Z}:x^{2}\leq 1\right\} $
}
\end{enumerate}
\feedback

\pregunta Sabent que $A=\left\{ a,\left\{ a\right\} ,\left\{ a,\left\{ a\right\}
\right\} \right\} $ i $B=\left\{ \left\{ a\right\} \right\} $, quina de les
següents afirmacions és falsa?

\begin{enumerate}
\opcio{a}{$B\subset A$
}
\opcio{b}{$a\in B$
}
\opcio{c}{$\left\{ a\right\} \subset A$
}
\opcio{d}{$\left\{ a,\left\{ a\right\} \right\} \in A$
}
\end{enumerate}
\feedback

\pregunta Donat el conjunt $A=\left\{ a,\left\{ a\right\} \right\} $, quina de
les següents afirmacions és falsa?

\begin{enumerate}
\opcio{a}{$\left\{ \left\{ \left\{ a\right\} \right\} \right\} \in \mathcal{P}(A)$
}
\opcio{b}{$\left\{ a\right\} \in\mathcal{P}(A)$
}
\opcio{c}{$\left\{ \emptyset\right\} \subset\mathcal{P}(A)$
}
\opcio{d}{$\left\{ a,\left\{ a\right\} \right\} \in\mathcal{P}(A)$
}
\end{enumerate}
\feedback

\pregunta Donat el conjunt referencial $E=\{1,2,3,4,5,6\}$ i els subconjunts $%
A=\left\{ x\in E:x\text{ és parell}\right\} $, $B=\left\{ x\in E:x\text{
és múltiple de 3}\right\} $ i $C=\left\{ x\in E:2\leq x\leq6\right\}
$, llavors

\begin{enumerate}
\opcio{a}{$A\cap B\cap C=\emptyset$
}
\opcio{b}{$\left( A\cup B\right) \cap C=C$
}
\opcio{c}{$\complement \left( A\cup B\cup C\right) =\left\{ 1\right\} $
}
\opcio{d}{$A\cup(B\cap C)=C$
}
\end{enumerate}
\feedback

\pregunta Donat el conjunt referencial $E$ i els subconjunts $A,B$ i $C$. En
simplificar l'expressió%
\begin{equation*}
\left[ (A\cap B)\cap C\right] \cup\left[ (A\cap B)\cap\complement C\right]
\cup\left( \complement A\cap B\right)
\end{equation*}
s'obté:

\begin{enumerate}
\opcio{a}{$A$
}
\opcio{b}{$B$
}
\opcio{c}{$C$
}
\opcio{d}{$E$
}
\end{enumerate}
\feedback

\pregunta En el conjunt dels nombres naturals es consideren les següents
relacions%
\begin{equation*}
\begin{array}{lll}
x~R_{1}~y & \Longleftrightarrow & x+y=10 \\
x~R_{2}~y & \Longleftrightarrow & x<y \\
x~R_{3}~y & \Longleftrightarrow & x,y\text{ són primers entre si}%
\end{array}%
\end{equation*}
Quina de les següents afirmacions és certa?

\begin{enumerate}
\opcio{a}{$R_{1}$ i $R_{3}$ són transitives
}
\opcio{b}{$R_{1}$ és simètrica i $R_{3}$ és antisimètrica
}
\opcio{c}{$R_{1}$ i $R_{3}$ són reflexives
}
\opcio{d}{$R_{2}$ és antisimètrica i $R_{3}$ és simètrica
}
\end{enumerate}
\feedback

\pregunta En el conjunt dels nombres reals es consideren les següents
relacions%
\begin{equation*}
\begin{array}{lll}
x~R_{1}~y & \Longleftrightarrow & x^{2}=y^{2} \\
x~R_{2}~y & \Longleftrightarrow & x(x+1)=y(y+1)%
\end{array}%
\end{equation*}
quina de les següents afirmacions és falsa?

\begin{enumerate}
\opcio{a}{$R_{1}$ i $R_{2}$ són relacions d'equivalència
}
\opcio{b}{La classe d'equivalència de $0$, segons $R_{1}$, és $\left[ 0%
\right] _{1}=\left\{ 0\right\} $ i, segons $R_{2}$, és $\left[ 0\right]
_{2}=\{0,-1\} $
}
\opcio{c}{La classe d'equivalència de $1$, segons $R_{1}$, és $\left[ 1%
\right] _{1}=\left\{ 1,-1\right\} $ i, segons $R_{2}$, és $\left[ 1%
\right] _{2}=\{1,-2\}$
}
\opcio{d}{$R_{1}$ no és una relació d'equivalència i $R_{2}$ sí
que ho és
}
\end{enumerate}
\feedback

\pregunta En el conjunt dels nombres naturals ordenat per la relació "ser
divisor de" es consideren els conjunts $A=\left\{ 1,2,3,4,5,6,7,8,9\right\} $ i
$B=\left\{ 3,4,6,12\right\} $. Llavors, quina de les següents afirmacions
és falsa?

\begin{enumerate}
\opcio{a}{$3$ i $4$ són elements minimals de $B$
}
\opcio{b}{Els elements maximals de $A$ són $6,8$ i $9$
}
\opcio{c}{$\sup A=2520$
}
\opcio{d}{$\max B=12$
}
\end{enumerate}
\feedback

\pregunta En el conjunt dels nombres reals ordenat segons la relació d'ordre
usual $\leq$ es considera el conjunt%
\begin{equation*}
A=\left\{ x\in\mathbb{R}:x^{2}+6x+5<0\right\}
\end{equation*}
Llavors, quina de les següents afirmacions és vertadera?

\begin{enumerate}
\opcio{a}{$\sup A=-1$
}
\opcio{b}{$-3$ és cota inferior d'$A$
}
\opcio{c}{$\max A=-1$
}
\opcio{d}{$\min A=-5$
}
\end{enumerate}
\feedback

\pregunta En el conjunt dels nombres enters es considera la relació següent%
\begin{equation*}
\begin{array}{ccc}
x~\equiv~y & \Longleftrightarrow & x-y\text{ és múltiple de }7%
\end{array}%
\end{equation*}
Si designem per $\left[ x\right] $ la classe de l'element $x$ segons $\equiv$, llavors quina de les següents afirmacions és vertadera?

\begin{enumerate}
\opcio{a}{$231\in\left[ 1\right] $
}
\opcio{b}{$\left[ -2\right] =\left[ 4\right] $
}
\opcio{c}{$-5\in \left[ 2\right] $
}
\opcio{d}{Cap de les anteriors
}
\end{enumerate}
\feedback
\end{enumerate}

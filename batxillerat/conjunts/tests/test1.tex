\begin{enumerate}
\pregunta Donat el conjunt $A=\left\{ -1,0,1\right\} $, quin dels seg\"{u}ents
conjunts coincideix amb $A$?

\begin{enumerate}
\opcio{a}{$\left\{ x\in\mathbb{N}:x^{3}-x=0\right\} $
}
\opcio{b}{$\left\{ x\in\mathbb{Q}:x^{2}\leq1\right\} $
}
\opcio{c}{$\left\{ x\in\mathbb{R}:x^{2}-1=0\right\} $
}
\opcio{d}{$\left\{ x\in \mathbb{Z}:x^{2}\leq 1\right\} $
}
\end{enumerate}
\feedback

\pregunta Sabent que $A=\left\{ a,\left\{ a\right\} ,\left\{ a,\left\{ a\right\}
\right\} \right\} $ i $B=\left\{ \left\{ a\right\} \right\} $, quina de les
seg\"{u}ents afirmacions \'{e}s falsa?

\begin{enumerate}
\opcio{a}{$B\subset A$
}
\opcio{b}{$a\in B$
}
\opcio{c}{$\left\{ a\right\} \subset A$
}
\opcio{d}{$\left\{ a,\left\{ a\right\} \right\} \in A$
}
\end{enumerate}
\feedback

\pregunta Donat el conjunt $A=\left\{ a,\left\{ a\right\} \right\} $, quin de
les seg\"{u}ents afirmacions \'{e}s falsa?

\begin{enumerate}
\opcio{a}{$\left\{ \left\{ a\right\} \right\} \subset \mathcal{P}(A)$
}
\opcio{b}{$\left\{ a\right\} \in\mathcal{P}(A)$
}
\opcio{c}{$\left\{ \emptyset\right\} \subset\mathcal{P}(A)$
}
\opcio{d}{$\left\{ a,\left\{ a\right\} \right\} \in\mathcal{P}(A)$
}
\end{enumerate}
\feedback

\pregunta Donat el conjunt referencial $E=\{1,2,3,4,5,6\}$ i els subconjunts $%
A=\left\{ x\in E:x\text{ \'{e}s parell}\right\} $, $B=\left\{ x\in E:x\text{
\'{e}s m\'{u}ltiple de 3}\right\} $ i $C=\left\{ x\in E:2\leq x\leq6\right\}
$, llavors

\begin{enumerate}
\opcio{a}{$A\cap B\cap C=\emptyset$
}
\opcio{b}{$\left( A\cup B\right) \cap C=C$
}
\opcio{c}{$\complement \left( A\cup B\cup C\right) =\left\{ 1\right\} $
}
\opcio{d}{$A\cup(B\cap C)=C$
}
\end{enumerate}
\feedback

\pregunta Donat el conjunt referencial $E$ i els subconjunts $A,B$ i $C$. En
simplificar l'expressi\'{o}%
\begin{equation*}
\left[ (A\cap B)\cap C\right] \cup\left[ (A\cap B)\cap\complement C\right]
\cup\left( \complement A\cap B\right)
\end{equation*}
s'obt\'{e}:

\begin{enumerate}
\opcio{a}{$A$
}
\opcio{b}{$B$
}
\opcio{c}{$C$
}
\opcio{d}{$E$
}
\end{enumerate}
\feedback

\pregunta En el conjunt dels nombres naturals es consideren les seg\"{u}ents
relacions%
\begin{equation*}
\begin{array}{lll}
x~R_{1}~y & \Longleftrightarrow & x+y=10 \\
x~R_{2}~y & \Longleftrightarrow & x<y \\
x~R_{3}~y & \Longleftrightarrow & x,y\text{ s\'{o}n primers entre si}%
\end{array}%
\end{equation*}
Quina de les seg\"{u}ents afirmacions \'{e}s certa?

\begin{enumerate}
\opcio{a}{$R_{1}$ i $R_{3}$ s\'{o}n transitives
}
\opcio{b}{$R_{1}$ \'{e}s sim\`{e}trica i $R_{3}$ \'{e}s antisim\`{e}trica
}
\opcio{c}{$R_{1}$ i $R_{3}$ s\'{o}n reflexives
}
\opcio{d}{$R_{2}$ \'{e}s antisim\`{e}trica i $R_{3}$ \'{e}s sim\`{e}trica
}
\end{enumerate}
\feedback

\pregunta En el conjunt dels nombres reals es consideren les seg\"{u}ents
relacions%
\begin{equation*}
\begin{array}{lll}
x~R_{1}~y & \Longleftrightarrow & x^{2}=y^{2} \\
x~R_{2}~y & \Longleftrightarrow & x(x+1)=y(y+1)%
\end{array}%
\end{equation*}
quina de les seg\"{u}ents afirmacions \'{e}s falsa?

\begin{enumerate}
\opcio{a}{$R_{1}$ i $R_{2}$ s\'{o}n relacions d'equival\`{e}ncia
}
\opcio{b}{La classe d'equival\`{e}ncia de $0$, segons $R_{1}$, \'{e}s $\left[ 0%
\right] _{1}=\left\{ 0\right\} $ i, segons $R_{2}$, \'{e}s $\left[ 0\right]
_{2}=\{0,-1\} $
}
\opcio{c}{La classe d'equival\`{e}ncia de $1$, segons $R_{1}$, \'{e}s $\left[ 1%
\right] _{1}=\left\{ 1,-1\right\} $ i, segons $R_{2}$, \'{e}s $\left[ 1%
\right] _{2}=\{1,-2\}$
}
\opcio{d}{$R_{1}$ no \'{e}s una relaci\'{o} d'equival\`{e}ncia i $R_{2}$ s\'{\i}
que ho \'{e}s
}
\end{enumerate}
\feedback

\pregunta En el conjunt dels nombres naturals ordenat per la relaci\'{o} "ser
divisor de" es consideren els conjunts $A=\left\{ 1,2,3,4,5,6,7,8,9\right\} $ i
$B=\left\{ 3,4,6,12\right\} $. Llavors, quina de les seg\"{u}ents afirmacions
\'{e}s falsa?

\begin{enumerate}
\opcio{a}{$3$ i $4$ s\'{o}n elements minimals de $B$
}
\opcio{b}{Els elements maximals de $A$ s\'{o}n $6,8$ i $9$
}
\opcio{c}{$\sup A=2520$
}
\opcio{d}{$\max B=12$
}
\end{enumerate}
\feedback

\pregunta En el conjunt dels nombres reals ordenat segons la relaci\'{o} d'ordre
usual $\leq$ es considera el conjunt%
\begin{equation*}
A=\left\{ x\in\mathbb{R}:x^{2}+6x+5<0\right\}
\end{equation*}
Llavors, quina de les seg\"{u}ents afirmacions \'{e}s vertadera?

\begin{enumerate}
\opcio{a}{$\sup A=-1$
}
\opcio{b}{$-3$ \'{e}s cota inferior d'$A$
}
\opcio{c}{$\max A=-1$
}
\opcio{d}{$\min A=-5$
}
\end{enumerate}
\feedback

\"{u}ent%

\opcio{b}{$\left[ -2\right] =\left[ 4\right] $
}

}
\opcio{d}{Cap de les anteriors
}
\end{enumerate}
\feedback
\end{enumerate}

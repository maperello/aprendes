\pregunta Quin és el conjunt de parts de $A=\{1,2\}$?

\begin{enumerate}
\opcio{a}{$\{\{1\},\{2\}\}$
}
\opcio{b}{$\{\emptyset,\{1\},\{2\},\{1,2\}\}$
}
\opcio{c}{$\{\{1,2\},\{1\}\}$
}
\opcio{d}{$\{\emptyset,\{1,2\}\}$
}
\end{enumerate}
\feedback

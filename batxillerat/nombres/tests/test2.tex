
\section{Test 2: Els nombres enters}

\iniciTest{tB}{a,b,d,c,a,d,b,d,d,a}

\begin{enumerate}

\pregunta En la construcció de \(\Z\), la relació d'equivalència sobre \(\N\times\N\) és:
\begin{enumerate}
\opcio{a}{\((a,b)\sim(c,d)\iff a+d=b+c\).}
\opcio{b}{\((a,b)\sim(c,d)\iff ab=cd\).}
\opcio{c}{\((a,b)\sim(c,d)\iff a+b=c+d\).}
\opcio{d}{\((a,b)\sim(c,d)\iff a=c\).}
\end{enumerate}
\feedback

\pregunta La classe \(\classe{(0,3)}\) representa habitualment:
\begin{enumerate}
\opcio{a}{\(3\).}
\opcio{b}{\(-3\).}
\opcio{c}{\(0\).}
\opcio{d}{\(1/3\).}
\end{enumerate}
\feedback

\pregunta La suma d'enters es defineix per:
\begin{enumerate}
\opcio{a}{\(\classe{(a,b)}+\classe{(c,d)}=\classe{(ac,bd)}\).}
\opcio{b}{\(\classe{(a,b)}+\classe{(c,d)}=\classe{(a+d,b+c)}\).}
\opcio{c}{\(\classe{(a,b)}+\classe{(c,d)}=\classe{(a-c,b-d)}\).}
\opcio{d}{\(\classe{(a,b)}+\classe{(c,d)}=\classe{(a+c,b+d)}\).}
\end{enumerate}
\feedback

\pregunta El producte \(\classe{(a,b)}\classe{(c,d)}\) és:
\begin{enumerate}
\opcio{a}{\(\classe{(ac,bd)}\).}
\opcio{b}{\(\classe{(a+c,b+d)}\).}
\opcio{c}{\(\classe{(ac+bd,ad+bc)}\).}
\opcio{d}{\(\classe{(ad,bc)}\).}
\end{enumerate}
\feedback

\pregunta L'oposat de \(\classe{(a,b)}\) és:
\begin{enumerate}
\opcio{a}{\(\classe{(b,a)}\).}
\opcio{b}{\(\classe{(a,b)}\).}
\opcio{c}{\(\classe{(0,0)}\).}
\opcio{d}{\(\classe{(a+1,b+1)}\).}
\end{enumerate}
\feedback

\pregunta Segons l'ordre definit en \(\Z\), \(\classe{(a,b)}\leq\classe{(c,d)}\) si:
\begin{enumerate}
\opcio{a}{\(a+b\leq c+d\).}
\opcio{b}{\(a+c\leq b+d\).}
\opcio{c}{\(ab\leq cd\).}
\opcio{d}{\(a+d\leq b+c\).}
\end{enumerate}
\feedback

\pregunta En \(\Z\), la solució de l'equació
\[
x+7=5
\]
és:
\begin{enumerate}
\opcio{a}{\(\classe{(7,5)}\).}
\opcio{b}{\(\classe{(5,7)}\).}
\opcio{c}{\(\classe{(12,0)}\).}
\opcio{d}{\(\classe{(0,12)}\).}
\end{enumerate}
\feedback

\pregunta Si \(a\leq b\) i \(c\leq 0\), aleshores:
\begin{enumerate}
\opcio{a}{\(ac\leq bc\).}
\opcio{b}{\(a+c\geq b+c\).}
\opcio{c}{\(ac=bc\).}
\opcio{d}{\(bc\leq ac\).}
\end{enumerate}
\feedback

\pregunta En la divisió euclidiana de \(-37\) entre \(5\), el quocient i el residu són:
\begin{enumerate}
\opcio{a}{\(q=-7,\ r=-2\).}
\opcio{b}{\(q=-7,\ r=2\).}
\opcio{c}{\(q=8,\ r=3\).}
\opcio{d}{\(q=-8,\ r=3\).}
\end{enumerate}
\feedback

\pregunta Quina afirmació de divisibilitat en \(\Z\) és certa?
\begin{enumerate}
\opcio{a}{\(-4\mid 20\).}
\opcio{b}{\(6\mid 20\).}
\opcio{c}{\(5\mid 23\).}
\opcio{d}{\(0\mid 12\).}
\end{enumerate}
\feedback

\end{enumerate}

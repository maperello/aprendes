\section{Test: Operacions amb fraccions de polinomis}

\iniciTest{pol3}{a,c,b,a,c,d,c,d,a,a}

\begin{enumerate}

\pregunta La fracció algebraica 
\begin{equation*}
\frac{x^{2}+x}{x^{2}+2x} 
\end{equation*}és equivalent a:

\begin{enumerate}[label=\alph*)]

\opcio{a}{$\frac{x+1}{x+2}$}

\opcio{b}{$\frac{x}{2x}$}

\opcio{c}{$\frac{x^{3}}{2x^{3}}$}

\opcio{d}{$\frac{x}{x+2}$}

\end{enumerate}

\feedback

\pregunta Quina és la fracció irreduïble, equivalent a\begin{equation*}
\frac{x^{2}-25}{x^{2}+10x+25} 
\end{equation*}

\begin{enumerate}[label=\alph*)]

\opcio{a}{$\frac{1}{10x}$}

\opcio{b}{$\frac{x+5}{x-5}$}

\opcio{c}{$\frac{x-5}{x+5}$}

\opcio{d}{Cap de les anteriors}

\end{enumerate}

\feedback

\pregunta En simplificar la fracció algebraica següent\begin{equation*}
\frac{2x^{4}+3x^{3}-4x^{2}-4x+3}{2x^{5}+x^{3}+2x^{2}-10x+5} 
\end{equation*}s’obté:

\begin{enumerate}[label=\alph*)]

\opcio{a}{$\frac{2x-3}{2x^{2}+5}$}

\opcio{b}{$\frac{2x+3}{2x^{2}+5}$}

\opcio{c}{$\frac{2x+3}{2x^{2}+1}$}

\opcio{d}{$\frac{2x-1}{2x^{2}+1}$}

\end{enumerate}

\feedback

\pregunta En efectuar les operacions indicades i simplificar\begin{equation*}
\frac{x+1}{2x-2}-\frac{4x}{x^{2}-1}-\frac{x^{2}+1}{1-x^{2}}-\frac{x-1}{2x+2} 
\end{equation*}s’obté:

\begin{enumerate}[label=\alph*)]

\opcio{a}{$\frac{x-1}{x+1}$}

\opcio{b}{$\frac{-1}{x+1}$}

\opcio{c}{$\frac{x}{x+1}$}

\opcio{d}{$\frac{x+1}{x-1}$}

\end{enumerate}

\feedback

\pregunta En efectuar les operacions indicades i simplificar\begin{equation*}
\left( \frac{x}{x+1}+\frac{x-1}{x}\right):\left( \frac{x}{x+1}-\frac{x-1}{x}\right) 
\end{equation*}s’obté:

\begin{enumerate}[label=\alph*)]

\opcio{a}{$-1$}

\opcio{b}{$1$}

\opcio{c}{$2x^{2}-1$}

\opcio{d}{$-2x^{2}+1$}

\end{enumerate}

\feedback
